## Automated note taking

Create a meeting notes template with the current date:

```javascript
import { Mac } from '@mac-sdk/automation';

async function createMeetingNotes() {
  const mac = new Mac();
  const today = new Date().toLocaleDateString();

  await mac.notes.iCloud.createNote({
    title: `Meeting Notes - ${today}`,
    body: `# Team Meeting - ${today}

## Agenda
1.
2.
3.

## Attendees
-

## Action Items
- [ ]
- [ ]`
  });

  await mac.notification('Notes Created', 'Your meeting notes are ready!');
}

createMeetingNotes().catch(console.error);
```

## Website screenshot tool

Open Safari and take screenshots using the Automation class:

```javascript
import { Mac, Automation, Key } from '@mac-sdk/automation';

async function captureScreenshots() {
  const mac = new Mac();
  const automation = new Automation();
  const timestamp = new Date().toISOString().replace(/[:.]/g, '-');

  const websites = [
    'https://github.com',
    'https://www.npmjs.com',
    'https://developer.apple.com'
  ];

  await mac.openApp('Safari');
  await automation.pause(1000);

  for (const url of websites) {
    // Navigate to URL bar and type address
    await automation.pressKeyCombo(Key.LeftSuper, Key.L);
    await automation.pause(200);
    await automation.type(url);
    await automation.pressKey(Key.Enter);

    // Wait for page to load
    await automation.pause(3000);

    // Take screenshot
    const domain = new URL(url).hostname.replace('www.', '');
    await mac.screenshot(`~/Desktop/${domain}-${timestamp}.png`);
  }

  await mac.notification(
    'Screenshots Complete',
    'All website screenshots have been saved to your Desktop'
  );
}

captureScreenshots().catch(console.error);
```

## Code snippet manager

Save code snippets to Apple Notes for later reference:

```javascript
import { Mac } from '@mac-sdk/automation';

class SnippetManager {
  constructor() {
    this.mac = new Mac();
  }

  async saveSnippet(name, code, language = 'javascript', description = '') {
    const content = `# ${name}

**Language**: ${language}
**Saved**: ${new Date().toLocaleString()}

${description ? `> ${description}\n\n` : ''}\`\`\`${language}
${code}
\`\`\``;

    await this.mac.notes.iCloud.createNote({
      title: `Snippet: ${name}`,
      body: content
    });
  }

  async listSnippets() {
    const results = await this.mac.notes.searchNotes('Snippet:');
    return results;
  }
}

// Example usage
async function main() {
  const manager = new SnippetManager();

  await manager.saveSnippet(
    'Array Shuffle',
    `function shuffle(arr) {
  for (let i = arr.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [arr[i], arr[j]] = [arr[j], arr[i]];
  }
  return arr;
}`,
    'javascript',
    'Shuffles an array in place using the Fisher-Yates algorithm.'
  );

  console.log('Code snippet saved to Notes!');
}

main().catch(console.error);
```

## Automated testing workflow

Open VS Code, run tests, and report results:

```javascript
import { Mac, Automation, Key } from '@mac-sdk/automation';

async function runTests() {
  const mac = new Mac();
  const automation = new Automation();

  try {
    // Open VS Code
    await mac.openApp('Visual Studio Code');
    await automation.pause(2000);

    // Open the integrated terminal
    await automation.pressKeyCombo(Key.LeftControl, Key.Grave);
    await automation.pause(1000);

    // Type the test command
    await automation.type('npm test');
    await automation.pressKey(Key.Enter);

    // Wait for tests to complete
    await automation.pause(5000);

    // Take a screenshot of the results
    await mac.screenshot('~/Desktop/test-results.png');

    // Notify with results
    await mac.notification(
      'Tests Completed',
      'Test results have been saved to your Desktop',
      'Glass'
    );
  } catch (error) {
    await mac.notification(
      'Test Error',
      'An error occurred while running tests',
      'Basso'
    );
    console.error('Test error:', error);
  }
}

runTests();
```

## Desktop automation with mouse and keyboard

Automate UI interactions using the `Automation` class:

```javascript
import { Mac, Automation, Key } from '@mac-sdk/automation';

async function automateDesktop() {
  const mac = new Mac();
  const automation = new Automation(500, 50);

  // Open Spotlight and search
  await automation.pressKeyCombo(Key.LeftSuper, Key.Space);
  await automation.pause(500);
  await automation.type('Calculator');
  await automation.pressKey(Key.Enter);
  await automation.pause(1000);

  // Click at specific coordinates
  await automation.click(300, 400);
  await automation.pause(200);

  // Scroll down
  await automation.scrollDown(500);

  // Take a screenshot of the result
  await mac.screenshot('~/Desktop/automation-demo.png');
  await mac.notification('Automation Complete', 'Desktop automation demo finished!');
}

automateDesktop().catch(console.error);
```

## System monitoring dashboard

Create a simple system monitoring tool using Mac SDK system controls:

```javascript
import { Mac } from '@mac-sdk/automation';

class SystemMonitor {
  constructor() {
    this.mac = new Mac();
    this.isMonitoring = false;
  }

  async startMonitoring(interval = 5000) {
    this.isMonitoring = true;
    console.log('Starting system monitoring...');

    while (this.isMonitoring) {
      const disk = await this.mac.system.getDiskSpace();
      const cpu = await this.mac.system.getCPUUsage();
      const memory = await this.mac.system.getMemoryUsage();

      console.clear();
      console.log('=== System Monitoring ===\n');
      console.log(`Disk: ${disk.used} used of ${disk.total} (${disk.percentage})`);
      console.log(`\nCPU:\n${cpu}`);
      console.log(`\nMemory:\n${memory}`);
      console.log('\nPress Ctrl+C to stop monitoring...');

      await new Promise(resolve => setTimeout(resolve, interval));
    }
  }

  stopMonitoring() {
    this.isMonitoring = false;
    console.log('\nMonitoring stopped');
  }
}

// Example usage
async function main() {
  const monitor = new SystemMonitor();

  process.on('SIGINT', () => {
    monitor.stopMonitoring();
    process.exit();
  });

  await monitor.startMonitoring(3000);
}

main().catch(console.error);
```

## Note change watcher

Watch a note for changes and react in real-time:

```javascript
import { Mac } from '@mac-sdk/automation';

async function watchNote() {
  const mac = new Mac();

  // First, create a note to watch
  await mac.notes.iCloud.createNote({
    title: 'Shared TODO',
    body: '# Shared TODO List\n\n- [ ] First task'
  });

  // Watch for changes (debounce: 3s, poll: every 2s)
  const cleanup = await mac.notes.iCloud.listenForChange(
    'Shared TODO',
    async (event) => {
      console.log(`Note "${event.title}" was modified`);
      console.log('Previous content:', event.previousContent);
      console.log('New content:', event.newContent);

      if (event.changes.added) {
        console.log('Added:', event.changes.added);
        console.log('At position:', event.changes.position);
        await mac.notification('Note Updated', `New content added to ${event.title}`);
      }

      if (event.changes.removed) {
        console.log('Removed:', event.changes.removed);
      }
    },
    3000,  // debounce: 3 seconds
    2000   // poll: every 2 seconds
  );

  console.log('Watching for changes... Edit the "Shared TODO" note in Apple Notes.');
  console.log('Press Ctrl+C to stop.');

  process.on('SIGINT', () => {
    cleanup();
    process.exit();
  });
}

watchNote().catch(console.error);
```
