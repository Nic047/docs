## Desktop automation with Automation class

The `Automation` class provides low-level mouse and keyboard control, separate from the `Mac` class. Use it for pixel-based UI automation.

```typescript
import { Mac, Automation, Key } from '@mac-sdk/automation';

const mac = new Mac();
const automation = new Automation(500, 50);

// Open an app and automate it
await mac.openApp('Safari');
await automation.pause(1000);

// Navigate to URL bar
await automation.pressKeyCombo(Key.LeftSuper, Key.L);
await automation.pause(200);

// Type a URL and press Enter
await automation.type('https://github.com');
await automation.pressKey(Key.Enter);
```

## Parallel task execution

Run multiple independent automation tasks in parallel using `Promise.all`:

```typescript
import { Mac } from '@mac-sdk/automation';

async function runParallelTasks() {
  const mac = new Mac();

  const tasks = [
    mac.notes.iCloud.createNote({
      title: 'Task 1',
      body: 'First parallel task'
    }),
    mac.system.screenshot('~/Desktop/parallel-1.png'),
    mac.system.createFile('parallel-output.txt', 'Created in parallel'),
  ];

  await Promise.all(tasks);

  console.log('All tasks completed!');
}
```

## Error handling and recovery

Implement robust error handling for production automations:

```typescript
class AutomationManager {
  private maxRetries = 3;

  async withRetry<T>(fn: () => Promise<T>, retries = this.maxRetries): Promise<T> {
    try {
      return await fn();
    } catch (error) {
      if (retries > 0) {
        console.warn(`Retrying (${retries} attempts left)...`);
        await new Promise(resolve => setTimeout(resolve, 1000));
        return this.withRetry(fn, retries - 1);
      }
      throw error;
    }
  }

  async safeAutomation() {
    const mac = new Mac();

    try {
      await this.withRetry(async () => {
        await mac.openApp('Safari');
      });
    } catch (error) {
      await mac.notification('Automation Failed', 'Please check the logs for details', 'Sosumi');
      throw error;
    }
  }
}
```

## Combining Mac and Automation

Use the `Mac` class for high-level operations and the `Automation` class for fine-grained UI control:

```typescript
import { Mac, Automation, Key } from '@mac-sdk/automation';

async function automatedWorkflow() {
  const mac = new Mac();
  const automation = new Automation();

  // High-level: open an app
  await mac.openApp('Visual Studio Code');
  await automation.pause(2000);

  // Low-level: interact with the UI
  await automation.pressKeyCombo(Key.LeftSuper, Key.N); // New file
  await automation.pause(500);
  await automation.type('console.log("Hello from Mac SDK!");');

  // High-level: take a screenshot and notify
  await mac.screenshot('~/Desktop/vscode-demo.png');
  await mac.notification('Done', 'Workflow complete!');
}
```

## State management

Manage state across automation runs by persisting to a file:

```typescript
import { Mac } from '@mac-sdk/automation';
import fs from 'fs/promises';
import path from 'path';

class StateManager {
  private statePath: string;
  private state: Record<string, any> = {};

  constructor(statePath?: string) {
    this.statePath = statePath || path.join(process.env.HOME || '~', '.mac-sdk/state.json');
  }

  async load() {
    try {
      const data = await fs.readFile(this.statePath, 'utf-8');
      this.state = JSON.parse(data);
    } catch {
      this.state = {};
    }
    return this;
  }

  async save() {
    await fs.mkdir(path.dirname(this.statePath), { recursive: true });
    await fs.writeFile(this.statePath, JSON.stringify(this.state, null, 2));
  }

  get<T = any>(key: string, defaultValue?: T): T {
    return (this.state[key] ?? defaultValue) as T;
  }

  set(key: string, value: any) {
    this.state[key] = value;
  }
}

// Usage
async function main() {
  const mac = new Mac();
  const state = await new StateManager().load();

  const lastRun = state.get('lastRun', 0);
  state.set('lastRun', Date.now());

  console.log(`Last run: ${new Date(lastRun).toLocaleString()}`);

  await state.save();
}
```

## Performance optimization

Optimize your automations for speed and reliability:

```typescript
import { Mac } from '@mac-sdk/automation';

class OptimizedAutomation {
  private cache = new Map<string, any>();

  constructor(private mac: Mac) {}

  async getCached<T>(key: string, ttl: number, fn: () => Promise<T>): Promise<T> {
    const cached = this.cache.get(key);
    const now = Date.now();

    if (cached && (now - cached.timestamp < ttl)) {
      return cached.value;
    }

    const value = await fn();
    this.cache.set(key, { value, timestamp: now });
    return value;
  }

  async batchOperations(operations: Array<() => Promise<void>>, batchSize = 5) {
    for (let i = 0; i < operations.length; i += batchSize) {
      const batch = operations.slice(i, i + batchSize);
      await Promise.all(batch.map(op => op()));
    }
  }
}
```

## Monitoring note changes

Use the `listenForChange` method to react to changes in Apple Notes:

```typescript
import { Mac } from '@mac-sdk/automation';

const mac = new Mac();

// Watch a note for changes (polls every 2 seconds, debounces for 5 seconds)
const cleanup = await mac.notes.iCloud.listenForChange(
  'My Note',
  (event) => {
    console.log('Note changed!');
    console.log('Added:', event.changes.added);
    console.log('Removed:', event.changes.removed);
    console.log('Note ID:', event.noteId);
  },
  5000,  // debounce time (ms)
  2000   // poll interval (ms)
);

// Stop watching when done
// cleanup();
```

These patterns help you build robust, maintainable, and efficient automations with Mac SDK.
