## File management

### Auto-organize downloads folder

```typescript
import { Mac } from '@mac-sdk/automation';
import path from 'path';

const mac = new Mac();
const DOWNLOADS = path.join(process.env.HOME || '~', 'Downloads');

const FOLDERS = {
  images: ['jpg', 'jpeg', 'png', 'gif', 'webp', 'svg'],
  documents: ['pdf', 'doc', 'docx', 'xls', 'xlsx', 'ppt', 'pptx'],
  archives: ['zip', 'rar', '7z', 'tar', 'gz'],
  code: ['js', 'ts', 'jsx', 'tsx', 'py', 'java', 'c', 'cpp', 'h', 'hpp'],
};

async function organizeDownloads() {
  const files = await mac.system.listDirectory(DOWNLOADS);

  for (const file of files) {
    if (file.isDirectory()) continue;

    const ext = path.extname(file.name).toLowerCase().slice(1);
    let targetFolder = 'other';

    for (const [folder, exts] of Object.entries(FOLDERS)) {
      if (exts.includes(ext)) {
        targetFolder = folder;
        break;
      }
    }

    const targetPath = path.join(DOWNLOADS, targetFolder);
    await mac.system.createDirectory(targetPath);

    const source = path.join(DOWNLOADS, file.name);
    const destination = path.join(targetPath, file.name);

    await mac.system.move(source, destination);
    console.log(`Moved ${file.name} to ${targetFolder}/`);
  }

  await mac.notification('Downloads Organized', `Processed ${files.length} files`);
}

// Run every hour
setInterval(organizeDownloads, 60 * 60 * 1000);
organizeDownloads().catch(console.error);
```

## System automation

### Smart workspace switcher

```typescript
import { Mac } from '@mac-sdk/automation';

const mac = new Mac();
const WORKSPACES = {
  work: {
    apps: ['Visual Studio Code', 'iTerm2', 'Slack'],
    volume: 30,
  },
  relax: {
    apps: ['Spotify', 'Safari'],
    volume: 60,
  },
};

async function switchWorkspace(workspaceName: string) {
  const workspace = WORKSPACES[workspaceName];
  if (!workspace) throw new Error(`Unknown workspace: ${workspaceName}`);

  // Close non-essential running apps
  const runningApps = await mac.apps.getRunningApps();
  for (const app of runningApps) {
    if (!workspace.apps.includes(app.name) && app.name !== 'Finder') {
      await mac.apps.killApp(app.name);
    }
  }

  // Open required apps
  for (const app of workspace.apps) {
    await mac.openApp(app);
  }

  // Adjust system volume
  await mac.system.setVolume(workspace.volume);

  await mac.notification('Workspace Activated', `Switched to ${workspaceName} mode`);
}
```

## Productivity

### Meeting note taker

```typescript
import { Mac } from '@mac-sdk/automation';

const mac = new Mac();

class MeetingNoteTaker {
  private currentMeeting: string | null = null;
  private meetingNotes = new Map<string, string>();

  async startMeeting(title: string, participants: string[] = []) {
    this.currentMeeting = title;

    const noteContent = `# ${title}

**Date**: ${new Date().toLocaleString()}
**Participants**: ${participants.join(', ') || 'None'}

## Agenda
1.
2.
3.

## Notes

## Action items
- [ ]
`;

    this.meetingNotes.set(title, noteContent);
    await mac.notes.iCloud.createNote({
      title: `Meeting: ${title}`,
      body: noteContent
    });

    await mac.notification('Meeting Started', `Recording notes for: ${title}`);
    return noteContent;
  }

  async addNote(note: string) {
    if (!this.currentMeeting) throw new Error('No active meeting');

    await mac.notes.iCloud.appendToNote(
      `Meeting: ${this.currentMeeting}`,
      `\n${note}`
    );
  }

  async endMeeting() {
    if (!this.currentMeeting) return;

    await mac.notification('Meeting Ended', `Notes saved for: ${this.currentMeeting}`);
    this.currentMeeting = null;
  }
}
```

## Development

### VS Code project scaffolding

```typescript
import { Mac } from '@mac-sdk/automation';

const mac = new Mac();

async function setupDevEnvironment(projectName: string) {
  // Create project structure
  await mac.system.createDirectory(`${projectName}/src`);
  await mac.system.createDirectory(`${projectName}/tests`);
  await mac.system.createDirectory(`${projectName}/docs`);

  // Create package.json
  const packageJson = {
    name: projectName,
    version: '1.0.0',
    description: `A project created with Mac SDK`,
    main: 'src/index.js',
    scripts: {
      test: 'jest',
      start: 'node src/index.js'
    }
  };

  await mac.system.writeFile(
    `${projectName}/package.json`,
    JSON.stringify(packageJson, null, 2)
  );

  // Create main file
  await mac.system.writeFile(
    `${projectName}/src/index.js`,
    `console.log('Hello, ${projectName}!');`
  );

  // Create README
  await mac.system.writeFile(`${projectName}/README.md`, `# ${projectName}

A project created with Mac SDK automation.

## Getting started

\`\`\`bash
npm install
npm start
\`\`\`
`);

  // Open in VS Code
  await mac.vscode.openFolder(projectName);

  await mac.notification('Project Created', `${projectName} is ready in VS Code!`);
}

setupDevEnvironment('my-app').catch(console.error);
```

### Automated browser testing

```typescript
import { Mac, Automation, Key } from '@mac-sdk/automation';

const mac = new Mac();
const automation = new Automation(500, 50);

async function runBrowserTest(url: string) {
  await mac.openApp('Safari');
  await automation.pause(1000);

  // Navigate to URL
  await automation.pressKeyCombo(Key.LeftSuper, Key.L);
  await automation.pause(200);
  await automation.type(url);
  await automation.pressKey(Key.Enter);
  await automation.pause(3000);

  // Take a screenshot
  const timestamp = new Date().toISOString().replace(/[:.]/g, '-');
  const screenshotPath = `~/Desktop/test-${timestamp}.png`;
  await mac.screenshot(screenshotPath);

  await mac.notification('Test Complete', `Screenshot saved for ${url}`);
  return screenshotPath;
}

runBrowserTest('https://example.com').catch(console.error);
```

## System management

### System info reporter

```typescript
import { Mac } from '@mac-sdk/automation';

const mac = new Mac();

async function generateSystemReport() {
  const disk = await mac.system.getDiskSpace();
  const cpu = await mac.system.getCPUUsage();
  const memory = await mac.system.getMemoryUsage();
  const wifi = await mac.system.getWiFiNetworks();
  const runningApps = await mac.apps.getRunningApps();

  const report = `# System Report

**Generated**: ${new Date().toLocaleString()}

## Disk space
- Total: ${disk.total}
- Used: ${disk.used} (${disk.percentage})
- Available: ${disk.available}

## Running apps (${runningApps.length})
${runningApps.map(app => `- ${app.name} (PID: ${app.pid})`).join('\n')}

## Wi-Fi networks
${wifi.map(n => `- ${n}`).join('\n')}
`;

  // Save to a note
  await mac.notes.iCloud.createNote({
    title: `System Report - ${new Date().toLocaleDateString()}`,
    body: report
  });

  await mac.notification('Report Generated', 'System report saved to Notes');
}

generateSystemReport().catch(console.error);
```

### Workspace cleanup

```typescript
import { Mac } from '@mac-sdk/automation';

const mac = new Mac();

async function cleanupDirectory(dirPath: string, options: {
  olderThanDays?: number;
  filePatterns?: string[];
  dryRun?: boolean;
} = {}) {
  const {
    olderThanDays = 30,
    filePatterns = ['*.tmp', '*.log', '*.cache'],
    dryRun = false
  } = options;

  const cutoffDate = new Date();
  cutoffDate.setDate(cutoffDate.getDate() - olderThanDays);

  let deletedCount = 0;
  let totalSize = 0;

  for (const pattern of filePatterns) {
    const files = await mac.system.findFiles(pattern, dirPath);

    for (const file of files) {
      const fullPath = `${dirPath}/${file}`;
      const stats = await mac.system.getInfo(fullPath);

      if (stats.mtime < cutoffDate) {
        if (dryRun) {
          console.log(`Would delete: ${fullPath} (${stats.size} bytes)`);
        } else {
          await mac.system.remove(fullPath);
          console.log(`Deleted: ${fullPath}`);
        }

        deletedCount++;
        totalSize += stats.size;
      }
    }
  }

  const message = dryRun
    ? `Would delete ${deletedCount} files (${(totalSize / 1024 / 1024).toFixed(2)} MB)`
    : `Deleted ${deletedCount} files (${(totalSize / 1024 / 1024).toFixed(2)} MB)`;

  await mac.notification('Cleanup Complete', message);
  return { deletedCount, totalSize };
}

// Dry run first
cleanupDirectory('/Users/me/temp', { dryRun: true }).catch(console.error);
```
